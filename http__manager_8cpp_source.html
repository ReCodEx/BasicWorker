92  curl_easy_setopt(curl, CURLOPT_SSL_VERIFYPEER, 1L);
93  curl_easy_setopt(curl, CURLOPT_SSL_VERIFYHOST, 2L);
94  // Throw exception on HTTP responses >= 400
95  curl_easy_setopt(curl, CURLOPT_FAILONERROR, 1L);

105  // Enable verbose for easier tracing

109 
110  // Close file

112
