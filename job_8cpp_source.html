76  throw job_exception("File server URL cannot be empty");
77  }

82 
83  // construct all tasks with their ids and check if they have all datas, but do not connect them
84  std::map<std::string, std::shared_ptr<task_base>> unconnected_tasks;
85  for (auto &task_meta : job_meta_->tasks) {
86  if (task_meta->task_id == "") {
87  throw job_exception("Task ID cannot be empty");
88  } else if (task_meta->priority == 0) {
89  throw job_exception("Priority cannot be zero");
90  } else if (task_meta->binary == "") {
91  throw job_exception("Command cannot be empty");
92  }
93 
94  // go through variables parsing
95  task_meta->binary = parse_job_var(task_meta->binary);
96  for (size_t i = 0; i < task_meta->cmd_args.size(); ++i) {
97  task_meta->cmd_args.at(i) = parse_job_var(task_meta->cmd_args.at(i));
98  }
99 
100  std::shared_ptr<task_base> task;
101 
102  // distinguish internal/external command and construct suitable object
103  if (task_meta->sandbox != nullptr) {
104 
105  // //////////////// //
106  // external command //
107  // //////////////// //
108 
109  auto sandbox = task_meta->sandbox;
110 
111  if (sandbox->name == "") {
112  throw job_exception("Sandbox name cannot be empty");
113  }
114 
115  // first we have to get appropriate hwgroup limits
116  std::shared_ptr<sandbox_limits> limits;
117  auto hwit = sandbox->loaded_limits.find(worker_config_->get_hwgroup());
118  if (hwit != sandbox->loaded_limits.end()) {
119  limits = hwit->second;
120  } else {
121  throw job_exception("Worker hwgroup was not found in loaded limits");
122  }
123 
124  // check and maybe modify limits
125  process_task_limits(limits);
126 
127  // go through variables parsing
128  limits->std_input = parse_job_var(limits->std_input);
129  limits->std_output = parse_job_var(limits->std_output);
130  limits->std_error = parse_job_var(limits->std_error);
131  limits->chdir = parse_job_var(limits->chdir);
132  std::vector<std::tuple<std::string, std::string, sandbox_limits::dir_perm>> new_bnd_dirs;
133  for (auto &bnd_dir : limits->bound_dirs) {
134  new_bnd_dirs.push_back(std::tuple<std::string, std::string, sandbox_limits::dir_perm>{
135  parse_job_var(std::get<0>(bnd_dir)), parse_job_var(std::get<1>(bnd_dir)), std::get<2>(bnd_dir)});
136  }
137  limits->bound_dirs = new_bnd_dirs;
138 
139  // ... and finally construct external task from given information
140  create_params data = {
141  worker_config_->get_worker_id(), id++, task_meta, limits, logger_, working_directory_.string()};
142 
143  task = factory_->create_sandboxed_task(data);
144 
145  } else {
146 
147  // //////////////// //
148  // internal command //
149  // //////////////// //
150 
151  task = factory_->create_internal_task(id++, task_meta);
152 
153  if (task == nullptr) {
154  throw job_exception("Unknown internal task: " + task_meta->binary);
155  }
156  }
157 
158  // add newly created task to container ready for connect with other tasks
159  unconnected_tasks.insert(std::make_pair(task_meta->task_id, task));

161

170  }
171

173  if (!task_queue_.empty() && task_queue_.at(0)->get_task_id() == "") {
174  task_queue_.erase(task_queue_.begin());
175  } else {
176  // something bad is happening here, stop this job evaluation
177  throw job_exception("Root task not present in first place after topological sort.");
178  }

255 void job::connect_tasks(
256  std::shared_ptr<task_base> root, std::map<std::string, std::shared_ptr<task_base>> &unconn_tasks)
257 {
258  for (auto &elem : unconn_tasks) {
259  const std::vector<std::string> &depend = elem.second->get_dependencies();
260 
261  // connect all suitable task underneath root
262  if (depend.size() == 0) {
263  root->add_children(elem.second);
264  elem.second->add_parent(root);
265  }
266 
267  for (size_t i = 0; i < depend.size(); ++i) {
268  try {
269  auto ptr = unconn_tasks.at(depend.at(i));
270  ptr->add_children(elem.second);
271  elem.second->add_parent(ptr);
272  } catch (std::out_of_range) {
273  throw job_exception("Non existing task-id (" + depend.at(i) + ") in dependency list");
274  }
275  }

279 std::vector<std::pair<std::string, std::shared_ptr<task_results>>> job::run()
280 {
281  std::vector<std::pair<std::string, std::shared_ptr<task_results>>> results;
282  progress_callback_->job_started(job_meta_->job_id);
283 
284  // simply run all tasks in given topological order
285  for (auto &task : task_queue_) {
286  // we don't want nullptr dereference
287  if (task == nullptr) {
288  continue;
289  }
290 
291  auto task_id = task->get_task_id();
292  try {
293  if (task->is_executable()) {
294  auto res = task->run();
295  logger_->info() << "Task \"" << task_id << "\" ran successfully";
296  progress_callback_->task_completed(job_meta_->job_id, task_id);
297 
298  // if task has some results than publish them in output
299  if (res != nullptr) {
300  results.push_back({task_id, res});
301  }
302  } else {
303  logger_->info() << "Task \"" << task_id << "\" marked as not executable, proceeding to next task";
304  progress_callback_->task_skipped(job_meta_->job_id, task_id);
305 
306  // even skipped task has its own result entry
307  std::shared_ptr<task_results> result(new task_results());
308  result->status = task_status::SKIPPED;
309  results.push_back({task_id, result});
310 
311  // we have to pass information about non-execution to children
312  task->set_children_execution(false);
313  }
314  } catch (std::exception &e) {
315  if (task->get_type() == task_type::INNER) {
316  // evaluation just encountered internal error and its quite possible
317  // that something is very wrong in here, so be gentle and crash like a sir
318  // and try not to mess up next job execution
319  throw;
320  }
321 
322  std::shared_ptr<task_results> result(new task_results());
323  result->status = task_status::FAILED;
324  result->error_message = e.what();
325  results.push_back({task_id, result});
326 
327  logger_->info() << "Task \"" << task_id << "\" failed: " << e.what();
328  progress_callback_->task_failed(job_meta_->job_id, task_id);
329 
330  if (task->get_fatal_failure()) {
331  logger_->info() << "Fatal failure bit set. Terminating of job execution...";
332  break;
333  } else {
334  // set executable bit in this task and in children
335  logger_->info() << "Task children will not be executed";
336  task->set_execution(false);
337  task->set_children_execution(false);
338  }
339  }
340  }
341 
342  progress_callback_->job_ended(job_meta_->job_id);
343  return results;
344 }

412 std::string job::parse_job_var(const std::string &src)
413 {
414  std::string res = src;
415 
416  size_t start = 0;
417  while ((start = res.find("${", start)) != std::string::npos) {
418  size_t end = res.find("}", start + 1);
419  size_t len = end - start - 2;
420  if (end == std::string::npos) {
421  throw job_exception("Not closed variable name: " + res.substr(start));
422  }
423 
424  if (job_variables_.find(res.substr(start + 2, len)) != job_variables_.end()) {
425  // we found variable and can replace it in string
426  res.replace(start, end - start + 1, job_variables_.at(res.substr(start + 2, len)));
427  }
428 
429  // start++; // just to be sure we're not in cycle
430  }
431 
432  return res;
433 }
