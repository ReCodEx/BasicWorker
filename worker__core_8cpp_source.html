42 void worker_core::run()
43 {
44  // connect broker_connection to real broker server
45  broker_->connect();
46  // start execution thread which will be receiving jobs
47  std::thread broker_thread;
48  logger_->info() << "Trying to create broker connection thread.";
49  try {
50  broker_thread = std::thread(std::bind(&broker_connection<connection_proxy>::receive_tasks, broker_));
51  } catch (std::system_error &e) {
52  logger_->emerg() << "Broker connection thread cannot be started: " << e.what();
53  return;
54  }
55  logger_->info() << "Broker connection thread created succesfully.";
56 
57  logger_->info() << "Job receiver will now start receiving.";
58  job_receiver_->start_receiving();
59 
60  broker_thread.join();
61  return;
62 }

123 void worker_core::log_init()
124 {
125  auto log_conf = config_->get_log_config();
126 
127  // Set up logger
128  // Try to create target directory for logs
129  auto path = fs::path(log_conf.log_path);
130  try {
131  if (!fs::is_directory(path)) {
132  fs::create_directories(path);
133  }
134  } catch (fs::filesystem_error &e) {
135  std::cerr << "Logger: " << e.what() << std::endl;
136  throw;
137  }
138 
139  // Create and register logger
140  try {
141  // Create multithreaded rotating file sink. Max filesize is 1024 * 1024 and we save 5 newest files.
142  auto rotating_sink =
143  std::make_shared<spdlog::sinks::rotating_file_sink_mt>((path / log_conf.log_basename).string(),
144  log_conf.log_suffix,
145  log_conf.log_file_size,
146  log_conf.log_files_count,
147  true);
148  // Set queue size for asynchronous logging. It must be a power of 2.
149  spdlog::set_async_mode(1048576);
150  // Make log with name "logger"
151  logger_ = std::make_shared<spdlog::logger>("logger", rotating_sink);
152  // Set logging level to debug
153  logger_->set_level(helpers::get_log_level(log_conf.log_level));
154  // Print header to log
155  if (helpers::compare_log_levels(spdlog::level::notice, logger_->level()) > 0) {
156  logger_->emerg() << "--- Started ReCodEx worker ---";
157  } else {
158  logger_->notice() << "------------------------------";
159  logger_->notice() << " Started ReCodEx worker";
160  logger_->notice() << "------------------------------";
161  }
162  } catch (spdlog::spdlog_ex &e) {
163  std::cerr << "Logger: " << e.what() << std::endl;
164  throw;
165  }
166 
167  return;
168 }
