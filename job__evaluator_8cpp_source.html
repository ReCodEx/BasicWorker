250 void job_evaluator::push_result()
251 {
252  logger_->info() << "Trying to upload results of job...";
253 
254  // just checkout for errors
255  if (job_ == nullptr) {
256  logger_->error() << "Pointer to job is null.";
257  return;
258  }
259 
260  // define path to result yaml file and archived result
261  fs::path result_yaml = results_path_ / "result.yml";
262  fs::path archive_path = results_path_ / "result.zip";
263 
264  logger_->info() << "Building yaml results file...";
265  // build yaml tree
266  YAML::Node res;
267  res["job-id"] = job_id_;
268  for (auto &i : job_results_) {
269  YAML::Node node;
270  node["task-id"] = i.first;
271  node["status"] = i.second->failed ? "FAILED" : "OK";
272  if (!i.second->error_message.empty()) {
273  node["error_message"] = i.second->error_message;
274  }
275 
276  auto &sandbox = i.second->sandbox_status;
277  if (sandbox != nullptr) {
278  YAML::Node subnode;
279  subnode["exitcode"] = sandbox->exitcode;
280  subnode["time"] = sandbox->time;
281  subnode["wall-time"] = sandbox->wall_time;
282  subnode["memory"] = sandbox->memory;
283  subnode["max-rss"] = sandbox->max_rss;
284 
285  switch (sandbox->status) {
286  case isolate_status::OK: subnode["status"] = "OK"; break;
287  case isolate_status::RE: subnode["status"] = "RE"; break;
288  case isolate_status::SG: subnode["status"] = "SG"; break;
289  case isolate_status::TO: subnode["status"] = "TO"; break;
290  case isolate_status::XX: subnode["status"] = "XX"; break;
291  }
292 
293  subnode["exitsig"] = sandbox->exitsig;
294  subnode["killed"] = sandbox->killed;
295  subnode["message"] = sandbox->message;
296 
297  node["sandbox_results"] = subnode;
298  }
299 
300  res["results"].push_back(node);
301  }
302 
303  // open output stream and write constructed yaml
304  std::ofstream out(result_yaml.string());
305  out << res;
306  out.close();
307  logger_->info() << "Yaml result file written succesfully.";
308 
309  // compress given result.yml file
310  logger_->info() << "Compression of results file...";
311  try {
312  archivator::compress(results_path_.string(), archive_path.string());
313  } catch (archive_exception &e) {
314  logger_->error() << "Results file not archived properly: " << e.what();
315  return;
316  }
317  logger_->info() << "Compression done.";
318 
319  // send archived result to file server
320  remote_fm_->put_file(archive_path.string(), result_url_);
321 
322  logger_->info() << "Job results uploaded succesfully.";
323  progress_callback_->job_results_uploaded(job_id_);
324  return;
325 }
